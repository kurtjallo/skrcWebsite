    **src/app/services/[slug]/page.tsx** -- Create the dynamic route page.

    This is a React Server Component. Structure:

    ```typescript
    import { notFound } from "next/navigation";
    import { servicePillars, getServiceBySlug } from "@/data/services";
    import { PillarDetail } from "@/components/services/PillarDetail";
    import type { Metadata } from "next";

    interface PageProps {
      params: Promise<{ slug: string }>;
    }

    // Generate static paths for all four pillars
    export async function generateStaticParams() {
      return servicePillars.map((pillar) => ({
        slug: pillar.slug,
      }));
    }

    // Dynamic metadata per pillar
    export async function generateMetadata({ params }: PageProps): Promise<Metadata> {
      const { slug } = await params;
      const pillar = getServiceBySlug(slug);
      if (!pillar) return { title: "Service Not Found" };

      return {
        title: `${pillar.name} | St Katharine Rural Connect`,
        description: pillar.whatHappens,
      };
    }

    export default async function ServicePillarPage({ params }: PageProps) {
      const { slug } = await params;
      const pillar = getServiceBySlug(slug);

      if (!pillar) {
        notFound();
      }

      return <PillarDetail pillar={pillar} />;
    }
    ```

    IMPORTANT: In Next.js 15, `params` is a Promise and must be awaited. Do NOT destructure params synchronously.
