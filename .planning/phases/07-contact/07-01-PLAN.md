    **Zod schema:**
    ```ts
    import { z } from "zod";

    export const contactSchema = z.object({
      name: z
        .string()
        .min(1, "We'd like to know what to call you")
        .max(100, "Name must be 100 characters or fewer"),
      email: z
        .string()
        .min(1, "We need your email so we can get back to you")
        .email("Please enter a valid email address so we can reach you"),
      phone: z
        .string()
        .max(20, "Phone number is too long")
        .optional()
        .or(z.literal("")),
      audience: z
        .enum(["farmer", "offshore-worker", "family", "community", "other"])
        .optional()
        .or(z.literal("")),
      message: z
        .string()
        .min(1, "Let us know how we can help")
        .max(5000, "Message must be 5000 characters or fewer"),
      honeypot: z.string().max(0, "Bot detected").optional(),
    });
    ```

    Key design decisions:
    - Empathetic error messages per the design spec ("We'd like to know what to call you" not "Name is required")
    - Phone is optional and accepts empty string (users can leave it blank)
    - Audience is optional and accepts empty string (not everyone self-identifies)
    - Honeypot field has max(0) -- any content means bot submission
    - Message max 5000 chars to prevent abuse without being too restrictive

    **TypeScript types derived from schema:**
    ```ts
    export type ContactFormData = z.infer<typeof contactSchema>;

    export type ContactFormState = {
      success: boolean;
      message: string;
      errors?: {
        [K in keyof ContactFormData]?: string[];
      };
    };
    ```

    ContactFormState is the server action return type. It includes:
    - `success`: whether submission succeeded
    - `message`: user-facing success/error message
    - `errors`: optional field-level error map (for failed validation)

    Export everything: `contactSchema`, `ContactFormData`, `ContactFormState`, `AUDIENCE_OPTIONS`.

    2. Create `src/app/contact/_actions/submit-contact.ts`:

    ```ts
    "use server";

    import { contactSchema, type ContactFormState } from "@/lib/schemas/contact";

    export async function submitContact(
      _prevState: ContactFormState,
      formData: FormData
    ): Promise<ContactFormState> {
      // Extract form fields
      const raw = {
        name: formData.get("name") as string,
        email: formData.get("email") as string,
        phone: formData.get("phone") as string,
        audience: formData.get("audience") as string,
        message: formData.get("message") as string,
        honeypot: formData.get("honeypot") as string,
      };

      // Honeypot check -- reject silently if filled
      // Return fake success to not alert bots
      if (raw.honeypot) {
        // Log for monitoring but return fake success
        console.log("[SPAM] Honeypot triggered:", {
          timestamp: new Date().toISOString(),
          name: raw.name,
          email: raw.email,
        });
        return {
          success: true,
          message: "Thank you for getting in touch. We'll get back to you as soon as we can.",
        };
      }

      // Server-side validation with zod
      const result = contactSchema.safeParse(raw);

      if (!result.success) {
        const fieldErrors: Record<string, string[]> = {};
        for (const issue of result.error.issues) {
          const field = issue.path[0] as string;
          if (!fieldErrors[field]) {
            fieldErrors[field] = [];
          }
          fieldErrors[field].push(issue.message);
        }

        return {
          success: false,
          message: "Please check the form below and try again.",
          errors: fieldErrors,
        };
      }

      // MVP: Log to console (no email service yet)
      console.log("[CONTACT] New submission:", {
        timestamp: new Date().toISOString(),
        name: result.data.name,
        email: result.data.email,
        phone: result.data.phone || "Not provided",
        audience: result.data.audience || "Not specified",
        message: result.data.message,
      });

      // TODO: Send email notification (e.g., Resend, SendGrid)
      // TODO: Store in database if needed

      return {
        success: true,
        message: "Thank you for getting in touch. We'll get back to you as soon as we can.",
      };
    }
    ```

    Key design decisions:
    - `"use server"` directive at top marks this as a Server Action
    - Function signature matches `useActionState` pattern: `(prevState, formData) => Promise<State>`
    - Honeypot check happens FIRST, before validation. Returns FAKE success to bots -- never reveals that the submission was rejected. This prevents bots from adapting.
    - Server validates with the SAME zod schema used client-side (single source of truth)
    - Field-level errors are extracted from zod issues and mapped to the field names
    - MVP logs to console only -- no email service integration needed yet
    - The `_prevState` parameter is prefixed with underscore since it's not used but is required by the `useActionState` API signature
